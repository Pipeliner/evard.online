bool tile_main_index(const char *str, tileidx_t *idx)
{
    ASSERT(str);
    if (!str)
        return false;
 
    string lc = str;
    for (unsigned int i = 0; i < lc.size(); i++)
        lc[i] = toalower(lc[i]);
 
    int num_pairs = sizeof(main_name_pairs) / sizeof(main_name_pairs[0]);
    bool result = binary_search<const char *, tileidx_t>(
       lc.c_str(), &main_name_pairs[0], num_pairs, &strcmp, idx);
    return result;
}
 
bool tile_main_equal(tileidx_t tile, tileidx_t idx)
{
    ASSERT_RANGE(tile, 0, TILE_MAIN_MAX);
    return idx >= tile && idx < tile + tile_main_count(tile);
}

{
    int num_pairs = sizeof(main_colour_pairs) / sizeof(main_colour_pairs[0]);
    tile_variation key(idx, col);
    tileidx_t found;
    bool result = binary_search<tile_variation, tileidx_t>(
       key, &main_colour_pairs[0], num_pairs,
       &tile_variation::cmp, &found);
    return (result ? found : idx);
}
